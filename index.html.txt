<script>
const botaoSim = document.getElementById("sim");
const botaoNao = document.getElementById("nao");
const caixa = document.getElementById("caixa");
const cupcake = document.getElementById("cupcake");

botaoSim.addEventListener("click", () => {
    document.querySelector("p").style.display = "none";
    document.querySelector(".botoes").style.display = "none";
    caixa.style.display = "block";
});

botaoNao.addEventListener("click", () => {
    alert("Ahh 😢... Tudo bem, Malu! O presente vai te esperar 🎁");
});

caixa.addEventListener("click", () => {
    caixa.style.display = "none";
    cupcake.style.display = "block";
    soltarConfete();
});

function soltarConfete() {
    for (let i = 0; i < 100; i++) {
        const confete = document.createElement("div");
        confete.classList.add("confete");
        confete.style.setProperty("--x", Math.random() * 100 + "vw");
        confete.style.setProperty("--hue", Math.random() * 360);
        document.body.appendChild(confete);

        setTimeout(() => confete.remove(), 3000);
    }
}
